  • Contexto de un evento 
                                    
let boton = document.getElementById("boton");
eventoAdjunto(boton, "click", EventHandler);

function eventoAdjunto(elemento, type, handler) {
    if (elemento.addEventListener) elemento.addEventListener(type, handler, false);
    else elemento.eventoAdjunto("on"+type, handler);
}

function EventHandler(e) {
    alert(this);
}
  • Contexto de metodos call(), bind(), apply() 
                                    
const persona = {
    name: 'Joel',
    surname: 'Sánchez',
    fullname: function() {
        const printFullname = function() { 
        console.log(`${this.name} ${this.surname}`);
        }; 
        return printFullname.bind(this);
    }
};
const printFullname = persona.fullname(); // bound
printFullname(); // `Joel Sánchez`

const libro = {
    paginaActual: 1,
    irPagina: function(page) { 
      this.paginaActual = page;
      alert(`Pagina Actual: ${this.paginaActual}`);
    }
}
const irPagina = libro.irPagina;
irPagina.call(libro, 10); // Pagina Actual: 10
irPagina.apply(libro, [10]); // Pagina Actual: 10
